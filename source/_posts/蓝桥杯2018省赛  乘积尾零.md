<p>注意：需要提交的是一个整数，表示末尾零的个数。不要填写任何多余内容。<br />
 </p>

<pre class="has">
<code class="language-cpp">#include&lt;bits/stdc++.h&gt;
using namespace std;
int a[11][11];

int main()
{
	int ans=0;
	long long t=1;
	for(int i=1;i&lt;11;i++)
	{
		for(int j=1;j&lt;11;j++)
		{
			cin&gt;&gt;a[i][j];
			int n=a[i][j];
			t=t*n;
			while(t&gt;0)
			{
				if(t%10==0)
				{
					ans++;
					t/=10;
				}
				else
				break;
			}
			t=t%10000;
		}
	}
	cout&lt;&lt;ans&lt;&lt;endl;
	return 0;
}
